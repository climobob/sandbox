# albedo.py
#
# To illustrate the "simplest climate model" documented at
# http://moregrumbinescience.blogspot.com/2012/03/return-of-simplest-climate-model.html

import matplotlib.pyplot as plt
import numpy as np

S = 1367        # solar constant
sigma = 5.67e-8 # Stefan-Boltzmann constant

def plot(T_central=255, T_width=5, max_albedo = 0.8, min_albedo = 0.3):
    """Shows the emission and absorption of radiation per square metre
    of surface.  To vary albedo or temperature Gaussian parameters,
    use the keyword arguments *T_central* (default 255), *T_width*
    (5), *max_albedo* (0.8) and *min_albedo* (0.3)."""
    T = np.arange(100,350,0.5)
    emit = sigma*(T**4)
    albedo_range = max_albedo-min_albedo
     # in Bob's blog piece there's a typo setting albedo_range to 0.3
    albedo = max_albedo - albedo_range*np.exp((-(T-T_central)**2)/(2*T_width*T_width))
    absorb = S*(1-albedo)/4
    plt.plot(T, emit,   label="emit")
    plt.plot(T, absorb, label="absorb")
    plt.legend()
    plt.show()
